async function loadIntel() { try { const res = await fetch('data/market_data.json?t=' + Date.now()); const data = await res.json(); const container = document.getElementById('display'); if (data.length > 0) { container.innerHTML = data.map(item => `
[ ${item.tag} ]

${item.title}

${item.desc}

${item.details}

View Full Operations Report →
`).join(''); } } catch (e) { console.error("Sync Error", e); } } setInterval(loadIntel, 10000); // Check for updates every 10 seconds loadIntel();
